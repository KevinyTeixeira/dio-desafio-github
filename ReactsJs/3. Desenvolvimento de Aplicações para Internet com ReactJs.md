import { INCREMENT, DECREMENT } from '/actions';

const initialState = {
    count: 0 // init state of our store
}

export function reducer(state = initialState, action) {
    switch(action.type) {
        case INCREMENT:
            return {
                count: state.count + 1
            };

                count: state.count - 1

    }
}
